\begin{small}
				\begin{itemize}
{\footnotesize
					\item Young's double slit experiment
				\item Discovered in 1801
				\item Light is a wave with amplitude, frequency and wavelength
				\item \textbf{Principle of superposition}: Waves superpose leading to constructive and destructive fringes.
				\item \href{https://javalab.org/en/youngs_double_slit_en/}{Animation for double slit experiment} [Click]
				}
			\end{itemize}
			%
\end{small}
		\end{column}
		\begin{column}[t]{0.38\textwidth}
%			\begin{center}
%				Particle
%			\end{center}
			%
			\begin{figure}
				\includegraphics[height=0.20\textheight]{qm_compton_effect}
			\end{figure}
\begin{small}
				\begin{itemize}
{\footnotesize
					\item Compton effect 
				\item Discovered in 1923.
				\item Light is a particle with momentum!
				\item \textbf{Conservation of momentum}: Light redshifts (wavelength increases) upon scattering off an electron.
				}
			\end{itemize}
\end{small}
		\end{column}
	\end{columns}
\end{frame}

	\begin{keyinsight}
		The name of the game is light-matter interaction. And nature plays the game in a \textbf{symmetric} fashion!
