\begin{frame}{Theories of quantum physics}
	\begin{itemize}
		\item Max Planck's postulate of quantization of light energy absorbed or emitted by electron in quanta (plural for quantum)
		\item Louis de Broglie's postulate of existence of matter waves
		\item Werner Heisenberg's principle of uncertainty
		\item Erwin Schr\"odinger's wave equation 
	\end{itemize}
\end{frame}

\begin{frame}{Wave-Particle Duality}
	%
\begin{columns}
	\begin{column}{0.8\textwidth}
			\begin{block}{de Broglie hypothesis}
		Any moving particle is associated with a wave. The waves associated with particles are called \textbf{de Broglie waves} or \textbf{matter waves}.
		%
	\end{block}
	\end{column}
	%
	\begin{column}{0.18\textwidth}
		\begin{figure}
			\includegraphics[width=\textwidth]{qm_matter_wave}
		\end{figure}
	\end{column}
\end{columns}
	\begin{columns}
		\begin{column}{0.85\textwidth}
			\begin{itemize}
				\item Louis de Broglie in 1924 postulated the existence of \textbf{matter waves}.
				\item He suggested since light wave exhibits particle like behaviour, matter particles should be expected to show wave-like properties. This is the hypothesis of \textbf{wave-particle duality}.
				\item The wavelength \(\lambda\) of the matter wave  is related to the momentum \(p\) of the matter particle by
				\[
				\lambda = \frac{h}{p}
				\] 
			\end{itemize}
		\end{column}

\begin{frame}{Matter wave: properties}
	%

	\begin{itemize}
		\item Matter waves are due to motion of particle and are independent of charge. Therefore, they are neither electromagnetic waves nor acoustic waves. They are a new kind of waves.
		\item Can propagate/travel through vacuum and do not require any medium for propagation.
		\item Smaller the mass, longer the de Broglie wavelength.
		\item Smaller the velocity, longer the de Broglie wavelength.
		\item Velocity of matter waves depends on velocity of particle and is not a constant quantity.
	\end{itemize}
