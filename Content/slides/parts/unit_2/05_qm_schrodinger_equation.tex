\begin{frame}{Schr\"odinger's wave equation}
	\begin{columns}
		\begin{column}{0.85\textwidth}
			\begin{block}{Wave equation hypothesis}
				If the\textbf{ wave function} of a particle of mass \( m \) is  \(\Psi(x, t)\), then it satisfies the \textbf{wave equation} given by
				\[
					-\frac{\hbar^2}{2m} \cdot \frac{\partial^2 \Psi(x, t)}{\partial x^2} + V(x) \Psi(x, t)= j \hbar \frac{\partial \Psi(x, t)}{\partial t}
				\]
				where \(V(x)\) is the potential function\footnote{Potential energy and not electric potential difference} and \(j\) is the imaginary constant \(\sqrt{-1}\).
			\end{block}
		\end{column}
		%
		\begin{column}{0.13\textwidth}
			\begin{figure}
				\begin{overpic}[width=1\textwidth]{Erwin_Schrodinger}
					% Overlay image on top
					\put(0,0){\includegraphics[width=0.4\textwidth]{Alfred_Nobel_1833_1896}}
				\end{overpic}
			\end{figure}
		\end{column}
	\end{columns}
		%
	\begin{itemize}
		\item Erwin Schr\"odinger postulated the wave equation in 1926 that incorporated principles of quanta introduced by Planck and wave-particle duality introduced by de Broglie.
		\item The wave function \(\Psi(x, t)\) describes the behaviour of particle
		\item \(\Psi(x, t)\) can be a \textbf{complex} quantity.
	\end{itemize}
\end{frame}
%
\begin{frame}{Analysis of the wave equation (W.E.)}
	%
	\[
	-\frac{\hbar^2}{2m} \cdot \frac{\partial^2 \Psi(x, t)}{\partial x^2} + V(x) \Psi(x, t)= j \hbar \frac{\partial \Psi(x, t)}{\partial t}
	\]
	\hrule
		\begin{enumerate}
		\item The independent variables/quantities are position \(x\) and time \(t\) whereas the dependent variable is the wave function \(\Psi\). Therefore, wave function is a function of position and time
		\[
		\Psi(x, t)
		\]
		\item W.E. is a \textbf{partial} differential equation.
		\begin{itemize}
				\item A differential equation relates dependent variable \(y\) to the independent variable \(x\). The relationship is given by derivatives \(\tfrac{\mathrm{d} \,y}{\mathrm{d} \, x},  \tfrac{\mathrm{d}^2 \, y}{\mathrm{d}\, x^2}, \ldots\)
			\item A partial differential equation relates dependent variable \(z\) to the independent variables \(x\) and \(y\). The relationship is given by partial derivatives \(\tfrac{\partial \,z}{\partial \, x}, \tfrac{\partial \,z}{\partial \, y},  \tfrac{\partial^2 \, z}{\partial\, x^2}, \ldots\)
		\end{itemize}
	\end{enumerate}
\end{frame}
%
%
\begin{frame}{Analysis of the wave equation (W.E.) contd...}
	%
	\[
	-\frac{\hbar^2}{2m} \cdot \frac{\partial^2 \Psi(x, t)}{\partial x^2} + V(x) \Psi(x, t)= j \hbar \frac{\partial \Psi(x, t)}{\partial t}
	\]
	\hrule
		\begin{enumerate}
			\setcounter{enumi}{3}
			\item W.E. is a \textbf{second order} differential equation.
			\begin{itemize}
				\item A first order differential equation has only first order derivative.
				\item A second order differential equation has upto second order derivatives.
			\end{itemize}
			\item The potential energy \(V(x)\) is a parameter whereas \(\hbar, m, j\) are constants.
		\end{enumerate}
\end{frame}
%
\begin{frame}{Solution of the W.E. : Separation of variables}
	%
	\[
	-\frac{\hbar^2}{2m} \cdot \frac{\partial^2 \colorbox{red}{$\Psi(x, t)$}}{\partial x^2} + \colorbox{green}{$V(x)$} \colorbox{red}{$\Psi(x, t)$}= j \hbar \frac{\partial \colorbox{red}{$\Psi(x, t)$}}{\partial t}
	\]
	\hrule
	\begin{itemize}
		\item The goal is to find the solution \colorbox{red}{$\Psi(x, t)$} for a given \colorbox{green}{$V(x)$}.
		\item We assume form of the solution as
		\[
			\Psi(x, t) = \colorbox{yellow}{$\psi(x)$} \colorbox{orange}{$\phi(t)$}
		\]
		where \(\psi(x)\) is a function of the position only and \(\phi(t)\) is a function of the time only.
		\item Upon substitution of above form, W.E. reduces to
		\[
		-\frac{\hbar^2}{2m} \colorbox{orange}{$\phi(t)$} \frac{\mathrm{d}^2 \colorbox{yellow}{$\psi(x)$}}{\mathrm{d} x^2}
		+
		\colorbox{green}{$V(x)$} \colorbox{yellow}{$\psi(x)$} \colorbox{orange}{$\phi(t)$}
		=
		j \hbar \colorbox{yellow}{$\psi(x)$}
		\frac{\mathrm{d} \colorbox{orange}{$\phi(t)$}}{\mathrm{d} t}
		\]
	\end{itemize}
\end{frame}
%
\begin{frame}{Solution of the W.E. : Separation of variables}
	%
	\begin{align*}
	\text{W.E.} \qquad
	-\frac{\hbar^2}{2m} \colorbox{orange}{$\phi(t)$} \frac{\mathrm{d}^2 \colorbox{yellow}{$\psi(x)$}}{\mathrm{d} x^2}
	+
	\colorbox{green}{$V(x)$} \colorbox{yellow}{$\psi(x)$} \colorbox{orange}{$\phi(t)$}
	&=
	j \hbar \colorbox{yellow}{$\psi(x)$}
	\frac{\mathrm{d} \colorbox{orange}{$\phi(t)$}}{\mathrm{d} t}
	\\
	\text{W.F.} \qquad  \qquad\qquad\qquad \qquad\qquad\qquad \qquad
	\Psi(x, t) &= \colorbox{yellow}{$\psi(x)$} \colorbox{orange}{$\phi(t)$}
	\end{align*}
	\hrule
	%
	\begin{itemize}
		\item Divide the above equation by total wave function so that
		\[
		-\frac{\hbar^2}{2m} \frac{1}{\colorbox{yellow}{$\psi(x)$}} \frac{\mathrm{d}^2 \colorbox{yellow}{$\psi(x)$}}{\mathrm{d} x^2}
		+
		\colorbox{green}{$V(x)$}
		=
		j \hbar \frac{1}{\colorbox{orange}{$\phi(t)$}}
		\frac{\mathrm{d} \colorbox{orange}{$\phi(t)$}}{\mathrm{d} t}
		\]
		\item Since L.H.S. is a function of \(x\) only and R.H.S. is a function of \(t\) only, each side must be equal to constant. Let us call this the \textbf{separation constant} \(\eta\).
		\[
		-\frac{\hbar^2}{2m} \frac{1}{\colorbox{yellow}{$\psi(x)$}} \frac{\mathrm{d}^2 \colorbox{yellow}{$\psi(x)$}}{\mathrm{d} x^2}
		+
		\colorbox{green}{$V(x)$}
		=
		\eta,
		\qquad
		j \hbar \frac{1}{\colorbox{orange}{$\phi(t)$}}
		\frac{\mathrm{d} \colorbox{orange}{$\phi(t)$}}{\mathrm{d} t}
		=
		\eta
		\]
	\end{itemize}
\end{frame}
%
\begin{frame}{Solution: Time dependent part of W.F}
	\[
	j \hbar \frac{1}{\colorbox{orange}{$\phi(t)$}}
	\frac{\mathrm{d} \colorbox{orange}{$\phi(t)$}}{\mathrm{d} t}
	=
	\eta
	\]
	\hrule
	%
	\begin{itemize}
		\item The time dependent side of W.E. is then given by
		\[
		\frac{\mathrm{d} \colorbox{orange}{$\phi(t)$}}{\mathrm{d} t}
		=
		-j \frac{\eta}{\hbar}  \colorbox{orange}{$\phi(t)$}
		\qquad
		\text{related to}
		\qquad
		\frac{\mathrm{d}\, y}{\mathrm{d}\, x} = c x
		\]
		\item The solution is a sinusoidal wave given by
		\[
		\colorbox{orange}{$\phi(t)$} = e^{-j \big(\frac{\eta}{\hbar} \big) t}
				\qquad
		\text{related to}
		\qquad
		y = e^{-j \omega t}
		\]
		with angular frequency \(\omega\) given by \hfill [\(\nu = \tfrac{\omega}{2\pi}\)]
		\[
		\omega = \frac{\eta}{\hbar}, \quad \Rightarrow \eta = \hbar \omega = \frac{h}{2\pi}\cdot 2\pi \nu = h \nu
		\]
		\item From Planck's law, \(E= h \nu \). Therefore,
		\[
		\eta = E,
		\qquad
		\text{and}
		\qquad
		\colorbox{orange}{$\phi(t)$} = e^{-j \big(\frac{E}{\hbar} \big) t} = e^{-j \omega t}
		\]
	\end{itemize}
\end{frame}
%
\begin{frame}{Solution: Time independent part of W.F}
	\vspace{-1em}
	\[
-\frac{\hbar^2}{2m} \frac{1}{\colorbox{yellow}{$\psi(x)$}} \frac{\mathrm{d}^2 \colorbox{yellow}{$\psi(x)$}}{\mathrm{d} x^2}
+
\colorbox{green}{$V(x)$}
=
\eta
\]
\hrule
%
\begin{itemize}
	\item The time independent portion can now be written as
	\[
	-\frac{\hbar^2}{2m} \frac{1}{\colorbox{yellow}{$\psi(x)$}} \frac{\mathrm{d}^2 \colorbox{yellow}{$\psi(x)$}}{\mathrm{d} x^2}
	+
	\colorbox{green}{$V(x)$}
	=
	E
	\]
	where the separation constant is replaced by the total energy \(E\).
	\item Therefore, the time independent part of W.E. is
	\[
	  \frac{\mathrm{d}^2 \colorbox{yellow}{$\psi(x)$}}{\mathrm{d} x^2}
	+
	\frac{2m}{\hbar^2}\big(E - \colorbox{green}{$V(x)$}\big)
	=
	0
	\]
	\item This is called the time independent Schr\"odinger wave equation. The aim is to solve for the time independent wave function \colorbox{yellow}{$\psi(x)$} for a given potential \colorbox{green}{$V(x)$}.
\end{itemize}
\end{frame}

\begin{frame}{Physical meaning of wave function}
	%
	\vspace{-1em}
	%
		\begin{columns}
		\begin{column}{0.85\textwidth}
			\begin{block}{Probability density hypothesis}
				If the\textbf{ wave function} of a particle is  \(\Psi(x, t)\), then the probability density function is given by
				\[
				|\Psi(x, t)|^2
				\]
			\end{block}
		\end{column}
		%
		\begin{column}{0.13\textwidth}
			\begin{figure}
				\begin{overpic}[width=1\textwidth]{Max_Born}
					% Overlay image on top
					\put(0,0){\includegraphics[width=0.4\textwidth]{Alfred_Nobel_1833_1896}}
				\end{overpic}
			\end{figure}
		\end{column}
	\end{columns}
	%
	\vspace{-1em}
\begin{small}
		\begin{itemize}
		\item Max Born in 1926 postulated the \textbf{interpretation} of wave function.
		\item Let us assume that we have solved for the time independent part of wave function. Thus,
		\[
		\Psi(x, t) = \psi(x) e^{-j \omega t}
		\]
		\item Since \(\Psi(x, t)\) is a complex function, it cannot represent physical quantity. Its modulus squared  is the \textbf{probability density} of finding particle between \(x\) and \(x+\mathrm{d}x\) and is given by
		\[
			|\Psi(x, t)|^2 = \Psi(x, t)\cdot \Psi^*(x, t) = \psi(x) e^{-j \omega t} \cdot \psi^*(x) e^{+j \omega t} = |\psi(x)|^2
		\]
		\item The probability of finding particle between \(x\) and \(x+\mathrm{d}x\) is then
		\[
			|\psi(x)|^2\cdot \mathrm{d}x
		\]
	\end{itemize}
\end{small}
\end{frame}
%
\begin{frame}{Boundary conditions}
	\begin{itemize}
		\item Total probability is unity.
		\[
		\int_{-\infty}^{\infty} |\psi(x)|^2 \, \mathrm{d}x = 1
		\]
		This condition is called \textbf{normality condition}.
		\item Continuity condition
		%
		\begin{enumerate}
			\item \(\psi(x)\) must be finite, single-valued and continuous.
			\item \(\tfrac{\partial\psi(x)}{\partial x}\) must be finite, single-valued and continuous.
		\end{enumerate}
		%
		\item \(\psi(x)\) must vanish at infinity.
		\[
		\lim_{x\rightarrow\infty} \psi(x) = 0, \qquad \lim_{x\rightarrow -\infty} \psi(x) = 0
		\]
		This condition is due to the \textbf{physicality condition}.
	\end{itemize}
	%
	\begin{problem}{Normalization of wave function}
		\begin{enumerate}
			\item Consider the wave function \(\Psi (x, t) = A \cos\big(\frac{\pi x}{2}\big) e^{-j \omega t}\) for \(-1 \leq x \leq +3\). Determine \(A\) so that \(\int_{-1}^{+3}|\Psi(x, t)|^2 \, \mathrm{d}x = 1\).
		\end{enumerate}
	\end{problem}
\end{frame}

		%
%\begin{small}
			The work of Planck, de Broglie, Heisenberg, Born, and Schr\"odinger can be summarized as
	%
	\vspace{-0.55em}
			\begin{enumerate}%[itemsep=0pt, topsep=0pt]
		\item The state of a particle is given by the wave function \(\Psi (x, t)\).
		\item The probability density of the particle is \(|\Psi (x, t)|^2\). The probability of finding the particle in between \(x\) and \(x+\mathrm{d}x\) is \(|\Psi (x, t)|^2 \mathrm{d}x\).
		\item If the uncertainty in position is \(\Delta x\) and uncertainty in position is \(\Delta p\), then
		\(
		\Delta x \, \Delta p \geq \frac{\hbar}{2}.
		\)
		\item The de Broglie wavelength of particle with momentum \(p\) is
		\(
		\lambda = \frac{h}{p}.
		\)
		\item The energy \(E\) of the particle with frequency \(\nu\) is
		\(
		E = h \nu.
		\)
	\end{enumerate}
