\begin{frame}[label=pre-M2T1-slot2]{Optoelectronics}
	%
	\begin{itemize}
		\item Electronics is the study of controlling electrical signals.
		\item Any electrical signal is characterized by its current and voltage properties.
		\item Electronics is broadly classified into three areas --
		\begin{itemize}
			\item \textbf{Microelectronics} -- here we study switching electrical signals, which is useful to represent binary logic. This is necessary for computation. This is also called \textbf{digital electronics}. Since power is not a concern, signals have currents of the order of microamperes (hence the name microelectronics) and voltages of the order of few volts 
			\item \textbf{Power electronics} -- here we study amplifying electrical signals to transfer energy. This is necessary for energy requirements of society as it is well known that electrical energy is efficient way of energy transfer. Signals have currents of the order of hundreds of amperes and voltages of the order of thousands of volts.  This is also called \textbf{analog electronics}.
			\item Opto-electronics~\footnote{Opto-electronics is a combination of optics and electronics} -- here we harness the interaction between electron and photon. The advantages of light like its high velocity, free source of energy (Sun light), are combined with the advantages of electron like its easy manipulation to develop devices of specific requirements. 
		\end{itemize}
	\end{itemize}
\end{frame}
%
\begin{frame}{Opto-electronic devices}
	\begin{definition}
		\begin{itemize}
			\item 	A electrical device is a electrical component with a characteristic or signature \(I-V\) behaviour.
			\item	An opto-electronic device is an electrical device whose \(I-V\) characteristic is determined by the interaction of light.
		\end{itemize}
	\end{definition}
	%
	\begin{itemize}
		\item The interaction of light with opto-electronic devices is of two-types --
		\begin{itemize}
			\item optical absorption
			\item Radiative recombination
		\end{itemize}
		%
		\item The opto-electronic devices based on optical absorption are
		\begin{itemize}
			\item solar cell
			\item photo-diode
		\end{itemize}
		\item whereas the opto-electronic devices based on radiative recombination are
		\begin{itemize}
			\item Light Emitting Diode (LED)
			\item Laser Diode
		\end{itemize}
	\end{itemize}
\end{frame}
%
\begin{frame}{Optical absorption}
	%
	\begin{definition}
\begin{itemize}
	\item Optical absorption is defined as the process of absorption of photon that leads to the generation of charge carriers.
	\item The generated electron-holes pairs are additional to the already present. Hence they are called \textbf{excess charge carriers}.
\end{itemize}
	\end{definition}
	%
	\begin{columns}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\includegraphics[width=\textwidth]{oe_sc_optical_absorption}
			\end{figure}
		\end{column}
		\begin{column}{0.58\textwidth}
				\begin{itemize}
			\item In semiconductors, photon with energy greater than energy band gap \(E_g\), it is absorbed and leads to generation of electron-hole pair.
		\end{itemize}
		%
		\end{column}
	\end{columns}
	%
	\begin{keyinsight}
		 In optical absorption, the photon is the source of energy.
	\end{keyinsight}
\end{frame}

	\begin{definition}
		Radiative recombination is defined as the process of generation of photon upon combination of excess charge carriers.

\begin{frame}[label=pre-M2T1-slot2]{Solar cell -- working principle}
	\begin{definition}
		The working principle of solar cell involves three processes--
		\begin{enumerate}
			\item Generation of \textbf{excess} electron-hole pairs upon the illumination of light, and
			\item Separation of electrons from holes \textbf{before} radiative recombination due to the built-in electric field.
			\item Generation of voltage due to \textbf{accumulation} of excess separated charge carriers at the ends of the pn junction
		\end{enumerate}
	\end{definition}
	%
	\vspace{-1em}
	%
	\begin{itemize}
		\item The solar cell pn junction is operated without bias i.e. zero bias.
		\item Upon illumination with photons of suitable energy, the photon are absorbed through the surface area. The covalent bonds of the lattice are broken releasing the free electrons and holes into the conduction band and valence band respectively.
	\end{itemize}
	%

	\end{problem}
\end{frame}

\begin{frame}[label=pre-M2T1-slot2]{Working principle -- 3. Photo-voltage}
	%
	\begin{definition}
		Photovoltage, also called photo e.m.f.~\footnote{e.m.f. stands for electro motive force} is the voltage appearing across the device due to the separation of excess charge carriers.
	\end{definition}
	%
	\vspace{-2em}
	%
	\begin{columns}
	\begin{column}[t]{0.4\textwidth}
		\begin{figure}
			\includegraphics[width=\textwidth]{oe_sc_photovoltage}
		\end{figure}
	\end{column}
	%
	\begin{column}[t]{0.58\textwidth}
		\begin{itemize}
			\item The swept excess electrons and holes \textbf{accumulate} at the n region and p region edges respectively
			\item The accumulation of positive charge at the p-edge and negative charge at the n-edge leads to the generation of a voltage \(V_L\).
		\end{itemize}
	\end{column}
\end{columns}
%
\begin{itemize}
	\item Since the source of voltage is illumination of photons, it is called \textbf{photovoltage}.
\end{itemize}
%
\begin{keyinsight}
	The photovoltage forward biases the pn junction.
\end{keyinsight}
\end{frame}

\begin{frame}{Solar cell -- Photocurrent}
	%
	\begin{definition}
		Photocurrent is the current flowing through the solar cell when it is connected across a load resistance.
	\end{definition}
	%
	\vspace{-2em}
	%
	\begin{columns}
		\begin{column}[t]{0.4\textwidth}
			\begin{figure}
				\includegraphics[width=\textwidth]{oe_sc_photocurrent}
			\end{figure}
		\end{column}
		%
		\pause
		%
		\begin{column}[t]{0.58\textwidth}
			\begin{itemize}
				\item Solar cell can be used to drive currents through loads.
				\item Upon connection to a load resistance \(R_L\), the accumulated holes at the p-edge flow anti-clockwise whereas the accumulated electrons at the n-edge flow clockwise.

			\end{itemize}
		\end{column}
	\end{columns}
	%
	\begin{itemize}
		\item Clockwise flow of holes leads to clockwise hole current \(I_h\) whereas anti-clockwise flow of electrons leads to clockwise electron current \(I_e\).
		\item The net current \(I_L\) is called photocurrent~\footnote{suffix L stands for light and \textbf{not load}.} and is given by 
		\(
		I_L = I_h + I_e.
		\)
	\end{itemize}
	%
	\begin{keyinsight}
		The photocurrent flows from n to p region.
	\end{keyinsight}
\end{frame}

\begin{frame}{Solar cell -- Power source}
	%
\begin{definition}
	Photocurrent is the current flowing through the solar cell when it is connected across a load resistance.
\end{definition}
%
\vspace{-2em}
%
\begin{columns}
	\begin{column}[t]{0.4\textwidth}
		\begin{figure}
			\includegraphics[width=\textwidth]{oe_sc_power_source}
		\end{figure}
	\end{column}
	%
	\begin{column}[t]{0.58\textwidth}
		\begin{itemize}
			\item The photo-voltage drop \(V_L\) is from p-n region.
			\item The photo-current \(I_L\) is from n-p region.
			\item Since the voltage drop is opposite the current direction across the solar cell, the power dissipated is negative.
			\item In other words, the \textbf{power generated is positive} and solar cell acts like a battery.
		\end{itemize}
	\end{column}
\end{columns}
%
\begin{keyinsight}
	Solar cell under illumination acts as a power source.
\end{keyinsight}
\end{frame}

	\begin{columns}

			\end{itemize}
		\end{column}
	\end{columns}

	Load forward biases the illuminated solar cell.
\end{keyinsight}

\begin{frame}[label=pre-M2T1-slot2]{Solar cell -- \(I-V\) characteristics}
	%
	\begin{columns}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\includegraphics[width=\textwidth]{oe_sc_i_v_characteristic}
			\end{figure}
		\end{column}
		%
		\pause
		%
		\begin{column}{0.58\textwidth}
			\begin{itemize}
				\item From Kirchoff voltage law, the photo-voltage is given by
				\[
				V_L = I_L R_L
				\]
				\item From diode \(I-V\) characteristics, the forward current \(I_F\) is given by
				\[
				I_F = I_s \left[ \exp\left(\frac{eV_L}{k_BT}\right) - 1\right]
				\]

			\end{itemize}
		\end{column}
	\end{columns}
	%
	\begin{itemize}
		\item Therefore the solar cell \(I-V\) characteristics is given by
		\begin{align*}
			I_\text{net} &= I_L - I_F 	\\
			&= I_L - I_s \left[ \exp\left(\frac{eV_L}{k_BT}\right) - 1\right]
		\end{align*}
	\end{itemize}
\end{frame}

\begin{frame}[label=pre-M2T1-slot2]{Solar cell -- short circuit current}
	%
	\begin{definition}
		Short circuit current is defined the current through solar cell when it is short circuited or zero load resistance is connected.
	\end{definition}
	%
		\begin{columns}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\includegraphics[width=\textwidth]{oe_sc_short_circuit_condition}
			\end{figure}
		\end{column}
		%
		\begin{column}{0.58\textwidth}
			\begin{itemize}
			\item When the load resistance is zero, the solar cell generates maximum current. This condition is called short circuit condition.
			\item At short circuit condition, \(R_L=0\), so the voltage drop across load is zero
			\[
			V_L = I_LR_L = 0
			\]
		\end{itemize}
		%
		\end{column}
	\end{columns}
	%
	\begin{itemize}
		\item The net current is the short circuit current \(I_\text{sc}\) and is given by
		\begin{align*}
			I_\text{sc} &= I_L - I_s \left(e^0 - 1\right)	\\
			&= I_L
		\end{align*}
	\end{itemize}
	%
\end{frame}
%
\begin{frame}[label=pre-M2T1-slot2]{Solar cell -- open circuit voltage}
	%
	\begin{definition}
		Open circuit voltage is defined the voltage across solar cell when it is open circuited or infinite load resistance is connected.
	\end{definition}
	%
	\vspace{-2em}
	%
	\begin{columns}
		\begin{column}[t]{0.4\textwidth}
			\begin{figure}
				\includegraphics[width=\textwidth]{oe_sc_open_circuit_condition}
			\end{figure}
		\end{column}
		%
		\begin{column}[t]{0.58\textwidth}
			\begin{itemize}
				\item When the load resistance is infinite, the solar cell generates maximum voltage. This condition is called open circuit condition.
				\item At open circuit condition, \(R_L=\infty\), so the current is zero
				\[
				I_L = \frac{V_L}{R} = 0
				\]
			\end{itemize}
			%
		\end{column}
	\end{columns}
	%
	\begin{itemize}
		\item The photovoltage is the open circuit voltage \(V_\text{oc}\) and is given by
		\begin{align*}
			I_L - I_s \left[ \exp\left(\frac{eV_\text{oc}}{k_BT}\right) - 1\right] = 0	 &\Rightarrow I_L = I_s \left[ \exp\left(\frac{eV_\text{oc}}{k_BT}\right) - 1\right] \\
			\exp\left(\frac{eV_\text{oc}}{k_BT}\right) = \frac{I_L}{I_s} + 1	&\Rightarrow  V_\text{oc} = \left(\frac{k_BT}{e}\right) \ln \left(1+ \frac{I_L}{I_s}\right)\\
		\end{align*}
	\end{itemize}
	%
\end{frame}
%
\begin{frame}{Solar cell -- Power optimization}
	%
	\begin{itemize}
		\item A solar cell is a power generating device. However, at the limiting conditions, the power generated is zero.
		\item At short circuit condition, the power generated is
		\[
		P_\text{sc} = V_\text{sc}I_\text{sc} = 0\cdot I_\text{sc} =0
		\]
		\item Similarly, at open circuit condition, the power generated is
		\[
		P_\text{oc} = V_\text{oc}I_\text{oc} = V_\text{oc} \cdot 0 = 0
		\]
		\item The electrical power generated is a function of drop across load resistance \(V = I R_L\) and is given by
		\[
		P = V \cdot I =  V \cdot \left[I_L - I_s \left\{ \exp\left(\frac{V_L}{V_t}\right) - 1 \right\} \right] ,
		\]
		where 
		\[
		V_t = \frac{k_B T}{e}
		\]
		is the thermal voltage associated with thermal energy.
	\end{itemize}
\end{frame}
%
\begin{frame}{Solar cell -- Power optimization}
	%
	\begin{itemize}
		\item To maximize \(P\) with respect to \(V\), we need to find \(V_m\), the voltage across the load when power is maximized. This is given by
		\begin{align*}
			\left.\frac{\mathrm{d} P}{\mathrm{d} V}  \right\|_{V_m}  &= 0 , \\
			\Leftrightarrow \left.I_L - I_s \left\{ \exp\left(\frac{V}{V_t}\right) - 1 \right\} - I_s \frac{V}{V_t}  \exp\left(\frac{V}{V_t}\right)\right\|_{V_m} &= 0, 	\\
			\Leftrightarrow I_L + I_s - I_s \exp\left(\frac{V_m}{V_t}\right) \left(1 + \frac{V_m}{V_t}\right) &=0,	\\
			\Leftrightarrow \left(1 + \frac{V_m}{V_t}\right) \exp\left(\frac{V_m}{V_t}\right) &= \frac{I_L + I_s}{I_s}.
		\end{align*}
		\[
		\therefore \left(1 + \frac{V_m}{V_t}\right) \exp\left(\frac{V_m}{V_t}\right) = 1 + \frac{I_L}{I_s}
		\]
		\item The value of \(V_m\) can be determined by trial and error.
	\end{itemize}
\end{frame}

\begin{frame}[label=pre-M2T1-slot2]{Output electrical power -- Fill factor}
	%
	\vspace{-2em}
	%
	\begin{columns}
		\begin{column}[t]{0.4\textwidth}
			\begin{figure}
				\includegraphics[width=\textwidth]{oe_sc_fill_factor}
				%
			\end{figure}
		\end{column}
		%
		\begin{column}[t]{0.58\textwidth}
			\begin{itemize}
				\item From the \(I-V\) characteristics of solar cell, it is evident that the maximum current generated cannot exceed the short circuit current \(I_\text{sc}\) and the maximum voltage generated cannot exceed the open circuit voltage \(V_\text{oc}\).
				\item The theoretically possible maximum power that can be generated should be less than the product of \(I_\text{sc}\) and \(V_\text{oc}\). In the \(I-V\) curve, this is represented as blue rectangle.


			\end{itemize}
		\end{column}
	\end{columns}

	\begin{itemize}
		\item The ratio of experimentally possible maximum power to theoretically possible maximum power is called Fill Factor and is given by 
		\[
		FF = \frac{V_m I_m}{V_\text{oc} I_\text{sc}}
		\]
	\end{itemize}

	\begin{keyinsight}
		Fill factor is a material property.

\begin{frame}[label=pre-M2T1-slot2]{Practical considerations -- Solar cell efficiency}
	%
	\begin{definition}
	 Conversion efficiency is defined as ratio of output electrical power to input optical power.
	\end{definition}
	\begin{itemize}
		\item To harness solar energy, the following practical limitations need to be considered.
		\item All the power cannot be harnessed as the optical absorption by a solar cell has a cut off wavelength due to the energy band gap.
		\item The photons with energy greater than energy band gap also generate electron-hole pairs. However, the difference of photon energy and band gap is dissipated as heat.
		\item The sum total of all these considerations is that a fraction of incident optical power is converted into electrical power and is given by the conversion efficiency \(\eta\) as
		\[
		\eta = \frac{P_\text{out}}{P_\text{in}}\times 100
		\]
	\end{itemize}
\end{frame}
