\begin{frame}{Luminescence}
	%
	\begin{definition}
		Luminescence refers to the general property of light emission.
	\end{definition}
	\begin{itemize}
		\item The previous two opto-electronic devices -- solar cell and photo diode -- were based on generation of excess carriers. The source of energy for generation was photon.
		\item The counterpart of generation is recombination.
		\item Recombination is of two types -- 
		\begin{itemize}
			\item radiative recombination -- the charge carriers recombine to emit photon
			\item non radiative recombination -- the charge carriers recombine to emit phonon i.e. heat. 
		\end{itemize}
		\item Here there are two types of radiative recombination --
		\begin{itemize}
			\item \textbf{Photo luminescence} -- If the excess charge carriers were generated by photon absorption, the photo emission from the recombination is called photo luminescence
			\item \textbf{Electro-luminescence} --  If the excess charge carriers were generated by current caused by an applied electric field, the photo emission from the recombination is called electro-luminescence.
		\end{itemize}
	\end{itemize}
\end{frame}

	%
	\begin{column}{0.68\textwidth}
		\begin{itemize}
			\item From the perspective of transfer of energy, the various processes can be understood as
\begin{small}
					\begin{itemize}
					\item \colorbox{pink}{Photo voltaic effect} -- Light \(\rightarrow\) Electrical
					\item \colorbox{green}{Photoluminescence} -- Light \(\rightarrow\) Light
					\item \colorbox{yellow}{Electroluminescence} -- Electrical \(\rightarrow\) Light
				\end{itemize}
\end{small}
			\item Processes that end up as light energy are due to \textbf{radiative recombination}.
			\item However, some other processes are also possible, where the energy ends up as heat
\begin{small}
					\begin{itemize}
					\item Photodiode -- Light \(\rightarrow\) Electrical \(\rightarrow\) Thermal
					\item pn junction at forward bias -- Electrical \(\rightarrow\) Thermal
					\item pn junction at reverse bias -- Thermal \(\rightarrow\) Electrical \(\rightarrow\) Thermal
				\end{itemize}
\end{small}
			\item Processes that end up as thermal energy are due to \textbf{non-radiative recombination}.
			\item The third opto-electronic device -- LED -- is based on electro-luminescence.
		\end{itemize}
		%
	\end{column}
\end{columns}
\end{frame}

\begin{frame}{Radiative vs Non radiative recombination}
	%
	\begin{figure}
		\includegraphics[width=0.8\textwidth]{oe_indirectvsdirect_defect_state}
	\end{figure}
	\begin{itemize}
		\item Defects are imperfections in semiconductor. These are different from impurities.
		\item Some examples of defects are absence of atom in lattice -- vacancy defect or presence of extra atom -- interstitial defect
		\item Defects arise when temperature is increased. These lead to defect levels \textbf{within} the forbidden band gap.
		\item If the crystal is without defects, radiative recombination is preferred and final particle emitted is a \textbf{photon}.
		\item If the crystal is with defects, electron and hole reach the defect level first, then they recombine non radiatively and final particle emitted is a \textbf{phonon} or lattice vibration. 
	\end{itemize}
	\begin{table}
		\begin{tabular}{lll}
			\toprule
			&	Radiative recombination	&	Non-radiative recombination	\\
			\midrule
			Final particle	&	Photon is emitted	&	Phonon is emitted	\\
			Defectless semiconductor	&	Easy	&	Difficult	\\
			Defect states 	&	Difficult	&	Easy	\\
			\bottomrule	
		\end{tabular}
	\end{table}
\end{frame}

\begin{frame}{Recombination lifetime}
	%
	\begin{definition}
		Recombination lifetime is the time period between generation and recombination of charge carriers.
	\end{definition}
	%
\begin{columns}
	\begin{column}{0.4\textwidth}
			\begin{figure}
		\includegraphics[width=\textwidth]{oe_indirectvsdirect_recombination_lifetime}
	\end{figure}
	\end{column}
	%
	\begin{column}{0.58\textwidth}
		\begin{itemize}
			\item Consider an electron-hole pair is generated. 
			\item When an electron is in conduction band, its energy has increased. It wants to recombine with a hole to reduce energy.
			\item The time taken to recombine radiatively \(\tau_\text{r}\) is less than the time taken to recombine non-radiatively \(\tau_\text{nr}\) since the defect level lowered the energy of electron.
		\end{itemize}
	\end{column}
\end{columns}
%
\end{frame}

\begin{frame}{Recombination rate -- Classification of semiconductors}
	%
	\begin{definition}
		Recombination rate is defined as the rate at which the excess charge carriers are recombining
	\end{definition}
	%
\begin{columns}
	\begin{column}{0.75\textwidth}
			\begin{itemize}
		\item The recombination rate is inversely related to the recombination lifetime.
		\item In the absence of defects, radiative recombination rate \(R_\text{r}\) is higher than non-radiative recombination rate \(R_\text{nr}\).
		\item However, as the defect density \(n_\text{defect}\) increases, the radiative recombination rate is not affected, but the non-radiative recombination rate increases.
	\end{itemize}
	\end{column}
	%
	\begin{column}{0.23\textwidth}
		\begin{figure}
			\includegraphics[width=\textwidth]{oe_indirectvsdirect_rate_with_defect_density}
		\end{figure}
	\end{column}
\end{columns}
%
\begin{itemize}
	\item The bandstructure of the semiconductor also affects the recombination rates. Based on bandstructure the semiconductors classified as --
	\begin{itemize}
		\item Direct band gap
		\item Indirect band gap
	\end{itemize}
\end{itemize}
\end{frame}
\begin{frame}{Direct band gap semiconductor}
	%
	\begin{definition}
		A direct band gap semiconductor is defined as a semiconductor with the top of valence band and bottom of conduction band at the \textbf{same} value of wave vector.
	\end{definition}
	%
	\vspace{-1em}
	%
	\begin{columns}
		\begin{column}{0.2\textwidth}
			\begin{figure}
				\includegraphics[width=\textwidth]{oe_indirectvsdirect_direct_bandgap}
			\end{figure}
		\end{column}
		%
		\begin{column}{0.78\textwidth}
			\begin{itemize}
				\item Consider electron moves from \(E_c\) to \(E_v\).
				\item In the radiative recombination process,
				\begin{itemize}
					\item Energy of electron is transferred to energy of photon.
					\item Initial momentum of electron is equal to final momentum.
					\item Therefore, both the energy and momentum of the (electron + hole + photon) system is \textbf{conserved}.
					\item \textbf{three} particles are involved
					\[
					\ce{e^-} + \ce{h^+} \rightarrow \text{photon} 
					\]
				\end{itemize}

			\end{itemize}
		\end{column}
	\end{columns}
	%
	\begin{keyinsight}
		Radiative type of recombination is the dominant mechanism in a direct bandgap semiconductor.
	\end{keyinsight}
\end{frame}
%
\begin{frame}{Indirect band gap semiconductor}
	%
	\begin{definition}
		An indirect band gap semiconductor is defined as a semiconductor with the top of valence band and bottom of conduction band at the \textbf{different} values of wave vector.
	\end{definition}
	%
	\begin{columns}
		\begin{column}{0.2\textwidth}
			\begin{figure}
				\includegraphics[width=\textwidth]{oe_indirectvsdirect_indirect_bandgap}
			\end{figure}
		\end{column}
		%
		\begin{column}{0.78\textwidth}
			\begin{itemize}
				\item Consider electron moves from \(E_c\) to \(E_v\).
				\item In the radiative recombination process,
				\begin{itemize}
					\item Energy of electron is transferred to energy of photon.
					\item Initial momentum of electron is not equal to final momentum.
					\item Therefore, energy of the (electron + hole + photon) system is \textbf{conserved} but the momentum is not conserved.

				\end{itemize}
				%

			\end{itemize}
		\end{column}
	\end{columns}
	%
	\begin{itemize}
		\item To conserve momentum, a \textbf{phonon assisted radiative recombination} occurs where difference in momentum is supplied by phonon and \textbf{four} particles are involved
		\[
		\ce{e^-} + \ce{h^+} \rightarrow \text{photon} + \text{phonon} 
		\]
	\end{itemize}
\end{frame}
%
\begin{frame}{Indirect band gap semiconductor}
	%
\begin{columns}
	\begin{column}{0.3\textwidth}
		\begin{figure}
			\includegraphics[width=\textwidth]{oe_indirectvsdirect_indirect_bandgap_non_radiative}
		\end{figure}
	\end{column}
\begin{column}{0.68\textwidth}
			\begin{itemize}
		\item Since the four particle process is difficult to occur, the phonon-assisted radiative recombination is not dominant mechanism.
		\item Instead the photon emission is avoided and the energy and momentum conservation is satisfied by emission of phonon only.
		\[
		\ce{e^-} + \ce{h^+} \rightarrow  \text{phonon} 
		\]
		%
		\item Emission of phonon leads to rise in temperature of material.
	\end{itemize}
\end{column}
\end{columns}
	\begin{keyinsight}
	Non-radiative type of recombination is the dominant mechanism in an indirect bandgap semiconductor.
\end{keyinsight}
\end{frame}
