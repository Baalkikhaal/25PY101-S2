\begin{frame}[label=pre-M2T1-slot1]{Summary of Classical free electron theory (CFET)}
	\begin{itemize}
		\item Derived $\sigma = \dfrac{ne^2\tau}{m}$.
		\item Connected microscopic electron properties to macroscopic Ohm’s law.
		\item Classical free electron theory has \colorbox{green}{qualitative and quantitative} \colorbox{green}{agreement with experiment values of conductivity.}
		\item However, the theory has drawbacks. Some of them are
		\begin{itemize}
			\item Incorrect prediction for conductivity vs valency.
			\item Cannot explain anomalous sign of Hall coefficient in some metals.
			\item Underestimation of mean free path.
			\item Cannot explain classification of materials into conductors. semi-conductors and insulators.
			\item Wrong prediction of conductivity vs temperature.
			\item \colorbox{red}{Overestimation of heat capacity.}
		\end{itemize} 
	\end{itemize}
\end{frame}

\begin{small}
		\begin{itemize}
		\item Proposed by Arnold Sommerfeld in 1927.
		\item Considered the \textbf{matter wave nature} of electron.
		\item Mutual repulsion between electrons is neglected i.e. electrons are independent -- \textbf{independent electron approximation}. 
		\item Assumed ``gas'' is \textbf{free} i.e. not under influence of  lattice --  \textbf{free electron approximation}.
		\item Electrons obey the \textbf{Pauli exclusion principle}.
		\item Role of lattice is to redistribute the energy distribution that obeys \textbf{quantum} Fermi-Dirac statistics --  quantum thermodynamics.
		\item The electron gas is called Fermi gas.
	\end{itemize}
\end{small}

\end{frame}

\begin{frame}[label=pre-M2T1-slot1]{Sommerfeld's Quantum Free electron theory -- Postulates}
	\begin{block}{Postulates}
		\begin{enumerate}
			\item \textbf{Waves:} Electrons are quantum waves with wavevector \(k\), angular frequency \(\omega\).
			\item \textbf{Fermion:} Electron is a spin \(\tfrac{1}{2}\) particle and obeys Pauli's exclusion principle.
			\item \textbf{Independent electron approximation:} Electrons are independent and mutual repulsion between them is ignored.
			\item \textbf{Free electron approximation:} Electrons are free and move in an infinite potential well.
			\item \textbf{Quantum Thermodynamics:} The thermalization is governed by Fermi-Dirac statistics.

		\end{enumerate}
	\end{block}
	%
\end{frame}

\begin{frame}{Quantum states in a cubic box}
	Consider a particle of mass $m$ confined in a cubic box of side $a$ with infinite walls.
	\begin{itemize}
		\item Wavefunction boundary conditions lead to quantum numbers $n_1,n_2,n_3\in\{1,2,3,\ldots\}$.
		\[
		\psi_{n_1,n_2,n_3}(x,y,z)
		= \sqrt{\frac{8}{a^{3}}}\,
		\sin\!\bigg(\frac{n_1\pi x}{a}\bigg)\,
		\sin\!\bigg(\frac{n_2\pi y}{a}\bigg)\,
		\sin\!\bigg(\frac{n_3\pi z}{a}\bigg),
		\]
		
		\item Energy levels are given by
		\[ E_{n_1,n_2,n_3} 
		= \frac{h^2}{8ma^2}\bigl(n_1^2+n_2^2+n_3^2\bigr). \]
	\end{itemize}
\end{frame}

\begin{frame}{Quantum degeneracy}
	%
	\begin{definition}
		\begin{itemize}
			\item Degeneracy is the condition when an energy level has more than one quantum state.
			\item If \(n\) quantum states have same energy level, then degeneracy of the energy level is \(n\).
			\item Degeneracy is also called multiplicity.
		\end{itemize}
	\end{definition}
	%
	\begin{problem}
		Five free electrons exist in a three dimensional potential well with all three widths equal to \(a=\SI{12}{\angstrom}\).
		%
		\begin{enumerate}
			\item Determine the Fermi energy level at \(T=\SI{0}{\kelvin}\).
			\item Repeat part 1 for 13 electrons.
		\end{enumerate}
	\end{problem}
	\begin{keyinsight}
		Quantum states can have the same energy level.
	\end{keyinsight}
\end{frame}

\begin{frame}{Quantum number space}
\begin{block}{Quantum number space}
	\begin{columns}
	\begin{column}{0.7\textwidth}

		\begin{itemize}
			\item A quantum state corresponds to a point with positive integer coordinates $(n_1,n_2,n_3)$ in the \textbf{number space}.
			\item This is an imaginary space.
			\item Quantum state lies in the positive octant of number space.
		\end{itemize}
	\end{column}
	%
	\begin{column}{0.28\textwidth}
		\includegraphics[width=\textwidth]{qt_number_space}
	\end{column}
\end{columns}
\end{block}
%
\vspace{-1em}
%
\begin{small}
		\begin{itemize}
		\item Let us count the number of states up to energy $E$.
		\item Define \(n\equiv\sqrt{n_1^2+n_2^2+n_3^2}\) (radial coordinate in \textit{n}-space).
		\item The number of states with $n$ less than some value $n_0$ equals the number of integer lattice points in the positive octant inside a sphere of radius $n_0$.
		\item We count the volume in \textit{n}-space:
		\[ N(n\le n_0) = g\times\frac{1}{8}\times \frac{4}{3}\pi n_0^3 = g\frac{\pi}{6} n_0^3, =  \]
		where $g$ is the degeneracy due to electron spin.
	\end{itemize}
\end{small}
\end{frame}
%	
\begin{frame}{Relate $n$ to energy}
	Using $E=(h^2/8ma^2)n^2$, solve for $n$:
	\[ n(E)=\frac{\sqrt{8ma^2E}}{h} .\]
	Thus the total number of states with energy less than or equal to $E$ is
	\begin{align*}
		N(E) &\approx g\frac{\pi}{6} n(E)^3
		= g\frac{\pi}{6}\left(\frac{\sqrt{8mL^2E}}{h}\right)^3
		= g\frac{\pi}{6}\left(\frac{\sqrt{8mE}}{h}\right)^3 L^3.
	\end{align*}
	Here, the volume $V=L^3$ and g=2.
\end{frame}
	
\begin{frame}[label=pre-M2T1-slot1]{Density of states $Z(E)=\dfrac{d(\tfrac{N}{V})}{dE}$}
	%
	\begin{definition}
\begin{itemize}
	\item Density of states \(Z(E)\)~\footnote{In Neamen, density of states is denoted by \(g(E)\).} is defined as the rate of change of the number of states per unit volume upto energy \(E\) with respect to energy \(E\).
	\item Therefore, the the number of states per unit volume from energy \(E\) to \(E+ \mathrm{d}E\) is given by
	\[
	Z(E)\, \mathrm{d}E.
	\]
\end{itemize}
	\end{definition}
	Number of states per unit volume is
	\[
	\rho_N(E) = \frac{N}{V} = \frac{\pi}{3}\left(\frac{\sqrt{8m}}{h}\right)^3 E^{3/2}
	\]
\end{frame}
	
\begin{frame}[label=pre-M2T1-slot1]{DOS in textbook form}
	Differentiate $\rho_N(E)$ with respect to $E$ to get the density of states function:
	\[
	Z(E)=\frac{\mathrm{d}\rho_N(E)}{\mathrm{d}E}
	= \frac{\pi}{2}\cdot \left(\frac{\sqrt{8m}}{h}\right)^3  E^{1/2}
	\]
	\[	
		Z(E) = \frac{\pi}{2}\cdot 8 \cdot  \left(\frac{2m}{h^2}\right)^{3/2} E^{1/2}
	\]
	%
	\begin{center}
		\framebox{$		\therefore Z(E) = 4\pi \cdot \left(\frac{2m}{h^2}\right)^{3/2} E^{1/2}.$}
	\end{center}
	%
	\begin{keyinsight}
		For 3D infinite potential, \(Z(E)\propto \sqrt{E}\).
	\end{keyinsight}
	%
	\begin{problem}
		Find the form of \(Z(E)\) for 2D infinite potential.
	\end{problem}
\end{frame}

\begin{frame}[label=pre-M2T1-slot1]{Fermi level}
	%
	\begin{definition}
		Fermi level of a metal is the maximum energy that an electron can have at \(T=\SI{0}{\kelvin}\).
	\end{definition}
	%
	\vspace{1em}
	%
\begin{columns}
	\begin{column}{0.8\textwidth}
			At absolute zero all states are filled up to the Fermi energy $E_F$.
	The total electron density $n_c$ (electrons per unit volume) is
	\begin{align*}
		n_c &= \frac{N}{V} = \int_0^{E_F}Z(E)\,\mathrm{d}E
		= 4\pi \cdot \left(\frac{2m}{h^2}\right)^{3/2}\int_0^{E_F}E^{1/2}\,\mathrm{d}E \\
		&= 4\pi \cdot \left(\frac{2m}{h^2}\right)^{3/2}\cdot \frac{2}{3}E_F^{3/2} \\
		&= \frac{8\pi}{3}\left(\frac{2m}{h^2}\right)^{3/2}E_F^{3/2}.
	\end{align*}
	\end{column}
	%
	\begin{column}{0.18\textwidth}
		\includegraphics[width=1.0\textwidth]{qt_fermi_level}
	\end{column}
\end{columns}
\end{frame}
	
\begin{frame}[label=pre-M2T1-slot1]{Relating Fermi energy to electron density}
	Invert the relation to get the Fermi energy as a function of density:
	%
	\[
				E_F^{3/2} = \left(\frac{h^2}{2m}\right)^{3/2}\cdot \frac{3n_c}{8\pi}.
	\]
	%
	%
	\begin{center}
		\framebox{$\therefore E_F = \frac{h^2}{2m}\,\left(\frac{3n_c}{8\pi}\right)^{2/3}.$}
	\end{center}

	This is the Fermi energy of a free electron gas in three dimensions at $T=0$.
	%
	\begin{problem}
		Calculate \(E_F\) for metal with electron density \(n_c = \SI{5.8e28}{\per\metre\cubed}\).
	\end{problem}
\end{frame}
